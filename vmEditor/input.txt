#include <string>
#include "vm.h"
using namespace std;

int main(int argc, char **argv){
	int i;
	string fileName;

	//we will ignore any flags for now
	for(i = 1; i < argc; ++i){
		if(argv[i][0] != '-'){
			fileName = string(argv[i]);
			break;
		}
	}

	Window w {fileName};
	w.init();
	return 0;
}
